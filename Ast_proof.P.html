Module Ast_proof.P

module P: Proof
type formula =
| TRUE
| FALSE
| Atom of string
| Reff of string
| Impl of formula * formula
| Equiv of formula * formula
| Not of formula
| Conj of formula * formula
| Dij of formula * formula
| Equal of formula * formula 
type proof =
| Refp of string
| Axiom of formula
| Asserted of formula
| AndElim of proof * formula
| MP of proof * proof * formula
| Rewrite of formula
| Unit of proof * proof list * formula
| Monotonicity of proof list * formula
| Trans of proof * proof * formula
| Hyp of formula
| Lemma of proof * formula 
val con : proof -> formula
type penv = (string, proof) H.t 
type fenv = (string, formula) H.t 
val inlinef : (string, formula) H.t ->
       formula -> formula
val inlinep : (string, proof) H.t ->
       (string, formula) H.t ->
       proof -> proof
type file =
| DeclF of string * formula * file
| DeclP of string * proof * file
| Proof of proof 
val extract : file -> proof
